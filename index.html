const SPREADSHEET_ID = 'https://script.google.com/macros/s/AKfycbyiyElI-ZJqbCzdPQnhxT7sQGhXdj2vsJppLMPFAe_fVr9P2_jYGo726SEATXAoVUsMXA/exec'; // ID вашей таблицы const ADDRESSES = ['Адрес1', 'Адрес2', 'Адрес3']; // Точные имена листов-адресов const PHOTOS_FOLDER = 'SchoolTrees_Photos'; // Корневая папка на Диске const HEADERS = [ 'UID', 'ID', 'Номер растения', 'Тип насаждения', 'Жизненная форма', 'Вид растения', 'Количество', 'Диаметр', 'Высота', 'Возраст', 'Состояние', 'Характеристика состояния', 'Рекомендации по уходу', 'Особо ценное', 'Remove', 'PhotoURL1', 'PhotoURL2', 'PhotoURL3', 'PhotoURL4' ]; function doGet(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); if (action === 'addresses') { return ContentService.createTextOutput(JSON.stringify({success: ADDRESSES})).setMimeType(ContentService.MimeType.JSON); } else if (action === 'list') { const address = e.parameter.address; if (!ADDRESSES.includes(address)) throw new Error('Invalid address'); const data = getSheetData(ss.getSheetByName(address)); return ContentService.createTextOutput(JSON.stringify({success: data})).setMimeType(ContentService.MimeType.JSON); } else if (action === 'tree') { const address = e.parameter.address; const uid = parseInt(e.parameter.uid); if (!ADDRESSES.includes(address)) throw new Error('Invalid address'); const data = getSheetData(ss.getSheetByName(address)); const row = data.find(r => parseInt(r.UID) === uid); return ContentService.createTextOutput(JSON.stringify({success: row || null})).setMimeType(ContentService.MimeType.JSON); } throw new Error('Unknown action'); } catch (error) { return ContentService.createTextOutput(JSON.stringify({error: error.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { try { const action = e.parameter.action; const ss = SpreadsheetApp.openById(SPREADSHEET_ID); if (action === 'updateRemove') { const address = e.parameter.address; const uid = parseInt(e.parameter.uid); const remove = e.parameter.remove === 'true'; if (!ADDRESSES.includes(address)) throw new Error('Invalid address'); updateRemove(ss.getSheetByName(address), uid, remove); return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON); } else if (action === 'uploadPhotos') { const address = e.parameter.address; const numStr = e.parameter.numPlant; const uid = parseInt(e.parameter.uid); if (!ADDRESSES.includes(address)) throw new Error('Invalid address'); const sheet = ss.getSheetByName(address); const folder = getOrCreatePhotosFolder(address, numStr, uid); const urls = uploadPhotos(e, folder); updatePhotoUrls(sheet, uid, urls); return ContentService.createTextOutput(JSON.stringify({success: urls})).setMimeType(ContentService.MimeType.JSON); } throw new Error('Unknown action'); } catch (error) { return ContentService.createTextOutput(JSON.stringify({error: error.toString()})).setMimeType(ContentService.MimeType.JSON); } } function getSheetData(sheet) { ensureHeaders(sheet); const range = sheet.getDataRange(); const values = range.getValues(); const data = []; for (let i = 1; i < values.length; i++) { const row = {}; HEADERS.forEach((header, col) => { row[header] = values[i][col] || ''; }); data.push(row); } // Сортировка по номеру растения как число data.sort((a, b) => parseFloat(a['Номер растения']) - parseFloat(b['Номер растения'])); return data; } function ensureHeaders(sheet) { const lastRow = sheet.getLastRow(); if (lastRow === 0) { sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); return; } const headerRow = sheet.getRange(1, 1, 1, HEADERS.length).getValues()[0]; const missing = HEADERS.filter(h => !headerRow.includes(h)); if (missing.length > 0 || headerRow[0] !== 'UID') { // Пересоздаем заголовки и нормализуем данные const dataRange = sheet.getDataRange(); const allValues = dataRange.getValues(); // Удаляем служебные строки (первые 5 или содержащие ${}) let startRow = 1; for (let i = 0; i < Math.min(5, allValues.length); i++) { if (String(allValues[i][0]).includes('${')) { startRow = i + 1; break; } } const cleanData = allValues.slice(startRow); // Создаем новую таблицу sheet.clear(); sheet.getRange(1, 1, 1, HEADERS.length).setValues([HEADERS]); // Маппинг старых колонок (предполагаем: ID в кол0 после служебных, Номер растения кол1 и т.д.) // Адаптируйте под вашу структуру Excel const newData = []; for (let i = 0; i < cleanData.length; i++) { const row = new Array(HEADERS.length).fill(''); row[0] = i + 1; // UID row[1] = cleanData[i][0] || ''; // ID row[2] = cleanData[i][1] || ''; // Номер растения // Заполните остальные по вашей структуре: row[3] = Тип..., etc. // Для примера оставляем пустыми newData.push(row); } if (newData.length > 0) { sheet.getRange(2, 1, newData.length, HEADERS.length).setValues(newData); } } } function updateRemove(sheet, uid, remove) { ensureHeaders(sheet); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === uid) { sheet.getRange(i + 1, HEADERS.indexOf('Remove') + 1).setValue(remove); break; } } } function getOrCreatePhotosFolder(address, numPlant, uid) { let root = getOrCreateFolder(PHOTOS_FOLDER); let addrFolder = root.getFoldersByName(address).next() || root.createFolder(address); let plantFolder = addrFolder.getFoldersByName(`${numPlant}_${uid}`).next() || addrFolder.createFolder(`${numPlant}_${uid}`); return plantFolder; } function getOrCreateFolder(name) { const folders = DriveApp.getFoldersByName(name); return folders.hasNext() ? folders.next() : DriveApp.createFolder(name); } function uploadPhotos(e, folder) { const urls = ['', '', '', '']; for (let i = 0; i < 4; i++) { const key = `photo${i}`; if (e.parameters[key] && e.parameters[key][0]) { const blob = Utilities.newBlob(Utilities.base64Decode(e.parameters[key][0].split(',')[1]), 'image/jpeg', `photo${i}.jpg`); const file = folder.createFile(blob); file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW); urls[i] = `https://drive.google.com/uc?export=view&id=${file.getId()}`; } } return urls; } function updatePhotoUrls(sheet, uid, urls) { ensureHeaders(sheet); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (parseInt(data[i][0]) === uid) { for (let j = 0; j < 4; j++) { const col = HEADERS.indexOf(`PhotoURL${j+1}`) + 1; sheet.getRange(i + 1, col).setValue(urls[j]); } break; } } } function fixAllAddressSheets() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); ADDRESSES.forEach(address => { const sheet = ss.getSheetByName(address); if (sheet) ensureHeaders(sheet); }); }
